szName属性 szText语法格式
line-type
线条类型 		solid, dot, dash, dash-dot, dash-dot-dot
font
字体 		font="字体唯一资源符"
text-offset
文本偏移量 		text-offset="5,10"
colors
颜色 		name="axis:0x124578;tickMajor:#235689"
axis :轴颜色
tickMajor : 长刻度颜色
tickMinor : 小刻度颜色
text : 文本颜色
gridline : 网格线颜色
back : 背景颜色
interval : 间隔颜色
scroll : 滚动滑块颜色
show-gridfill
显示网格填充 		true / false
show-gridline
显示网格线 		true / false
Precision
刻度文本显示精度 		Precision="2"
Scientific
刻度文本以科学计数法显示 		Scientific="true"
AlignTextSign
刻度文本正负符号对齐 		默认为false. AlignTextSign="true"
